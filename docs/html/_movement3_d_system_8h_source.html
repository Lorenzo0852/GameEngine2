34 void Run(float deltaTime)
35 {
36 for (auto& entity : GetSystemEntities())
37 {
38 std::shared_ptr<glt::Node> node = entity.GetComponent<Node3DComponent>().node;
39 auto& transform = entity.GetComponent<TransformComponent>();
40 const auto& rigidbody = entity.GetComponent<RigidbodyComponent>();
41
42 node->translate(glm::vec3(rigidbody.velocity.x * deltaTime, rigidbody.velocity.y * deltaTime, rigidbody.velocity.z * deltaTime));
43
44 transform.rotation.x += static_cast<float>(rigidbody.angularVelocity.x * deltaTime);
45 transform.rotation.y += static_cast<float>(rigidbody.angularVelocity.y * deltaTime);
46 transform.rotation.z += static_cast<float>(rigidbody.angularVelocity.z * deltaTime);
47
48 node->rotate_around_x(rigidbody.angularVelocity.x * deltaTime);
49 node->rotate_around_y(rigidbody.angularVelocity.y * deltaTime);
50 node->rotate_around_z(rigidbody.angularVelocity.z * deltaTime);
51
52 glm::vec3 translation = glt::extract_translation(node->get_transformation());
53 transform.position = glm::vec3(translation.x, translation.y, translation.z);
54 }
55 }
