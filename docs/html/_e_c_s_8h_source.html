79 /****************************************************************************************\
80 * COMPONENT
81 \****************************************************************************************/
82 /*
83 * The compiler will generate different classes per type TComponent. So being this the case,
84 * everytime a new class is generated (say Component_TransformComponent for Component<TransformComponent>,
85 * we can get an unique ID using its static method GetId().
86 *
87 * A static member is shared by all objects of the same class, meaning that GetId() will always return
88 * the same ID for the same Component class.
89 *
90 * Local static variables are initialized ONLY the first time control enters the function, so its ID doesn't
91 * keep getting incremented using nextId++ while on the same component type. A new component type will, however
92 * enter control of the function again (as it is essentially a new class created by the compiler), incrementing
93 * the ID and being unique to that new class (new component type).
94 *
95 * More on static members: https://en.cppreference.com/w/cpp/language/static
96 */
97
98
99 struct IComponent
100 {
101 protected:
102 static int nextId;
103 };
104
108 template <typename TComponent>
109 class Component : public IComponent {
110 public:
115 static unsigned GetId() {
116 static auto id = nextId++;
117 return id;
118 }
119 };

311 template <typename TComponent, typename ...TArgs>
312 void Registry::AddComponent(Entity entity, TArgs&& ...args)
313 {
314 const auto componentId = Component<TComponent>::GetId();
315 const auto entityId = entity.GetId();
316
317 //We resize the pool if needed
318 if (componentId >= componentPools.size())
319 {
320 componentPools.resize(componentId + 1, nullptr);
321 }
322
323 //We check if there is any component in the index we are checking. If there isn't, create it.
324 if (!componentPools[componentId])
325 {
326 // Only deallocate this when all the references go out of scope
327 // It's a shared_ptr as we immediately assign it to a different variable, with a different scope.
328 // The object of the shared ptr will destroy when everything that makes reference to it goes out of scope.
329 std::shared_ptr<Pool<TComponent>> newComponentPool = std::make_shared<Pool<TComponent>>();
330 componentPools[componentId] = newComponentPool;
331 }
332
333 std::shared_ptr<Pool<TComponent>> componentPool = std::static_pointer_cast<Pool<TComponent>>(componentPools[componentId]);
334
335 if (entityId >= componentPool->GetSize())
336 {
337 componentPool->Resize(numEntities);
338 }
339
340 TComponent newComponent(std::forward<TArgs>(args)...);
341 componentPool->Set(entityId, newComponent);
342 entityComponentSignatures[entityId].set(componentId);
343 }

398 template <typename TSystem>
399 TSystem& Registry::GetSystem() const
400 {
401 auto system = systems.find(std::type_index(typeid(TSystem)));
402 //.find() returns a pair, where ->first is the first element and ->second the second one.
403 //We need to dereference it so it doesn't return a memory address, but its content.
404 return *(std::static_pointer_cast<TSystem>(system->second));
405 }
